async function executarBuscaEEnviar() { const webhook = document.getElementById('n8n_url').value; const termo = document.getElementById('termo').value; const bairro = document.getElementById('bairro').value; const cidade = document.getElementById('cidade').value; const estado = document.getElementById('estado').value; const max = document.getElementById('maxres').value; if(!webhook || !termo || !cidade) return alert("Preencha o Webhook, o termo e a cidade!"); const btn = document.getElementById('btn'); btn.disabled = true; btn.innerText = "ENVIANDO COMANDO..."; log(`Enviando ordem de busca para o Agente n8n...`); try { const n8n = await fetch(webhook, { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ termo: termo, bairro: bairro, cidade: cidade, estado: estado, limite: max }) }); if(n8n.ok) log("✅ SUCESSO: O Agente n8n recebeu o comando e vai iniciar a busca e os disparos!"); else log("❌ ERRO: O n8n não respondeu."); } catch (e) { log("❌ ERRO DE CONEXÃO: Verifique se o n8n está ativo."); } btn.disabled = false; btn.innerText = "BUSCAR E ENVIAR PARA AGENTE →"; }
